docker run --rm -d --name postgres --shm-size=256m \
-p 5432:5432 -e POSTGRES_PASSWORD=postgres -e PGDATA=/tmp \ 
-v ~/docker-mnt/postgres:/var/lib/postgresql/data postgres;

docker run -it --name psql --link postgres:postgres \
--rm postgres sh -c 'psql -h postgres -U postgres -d postgres';

--link postgres:postgres --rm postgres sh -c 'psql -h postgres -U postgres -d cdc';
